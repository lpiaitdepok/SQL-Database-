sqlite3

.help

sqlite path/database_name.ext

# open database
.open path/database_name.ext

# show databases
.database

# exit from sqlite3
.exit

# show tables
.tables

# show table
.table 'table_name'

# show table structure
.schema 'table_name'

# show full table structure
.fullschema

# show indexes
.indexes
